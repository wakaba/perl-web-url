Web::URL::Canonicalize - Perl URL Canonicalizer

  use Web::URL::Canonicalize;
  $url = '//テスト.example?ほげ#ふが';
  $canon_base_url = 'https://foo.test/';
  $charset = 'euc-jp';
  $canon_url = url_to_canon_url $url, $canon_base_url, $charset;
  is $canon_url, 'https://xn--zckzah.example/?%A4%DB%A4%B2#ふが';
